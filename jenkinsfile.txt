#!groovy
pipeline{
    environment{
        registry = 'abhisheksg02/task-assign'
        registryCredentials = 'docker-credentials'
    }
    agent any
    stages{
        stage('clone git repository'){
            steps{
                git credentialsId: 'git-credentials', url: 'https://github.com/Abhisheksg02/rps-ant.git'
            }
        }
        stage('Build a docker image'){
            steps{
                sh 'echo Build a docker image'
                script {
                    image = docker.build("${registry}:$BUILD_NUMBER")
                }
            }
        }
        stage('Push Docker image to a registry'){
            steps{
                sh 'echo push to registry'
                script {
                    docker.withRegistry('',registryCredentials) {
                    image.push()
                    image.push('latest')
                    }
                }
            }
        }
    }
}
